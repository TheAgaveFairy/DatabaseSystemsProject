1. Total Sales per Month
This query calculates the total revenue generated each month, using the Invoice and Transactions tables.

SELECT 
    EXTRACT(YEAR FROM i.INVOICE_DATE) AS year,
    EXTRACT(MONTH FROM i.INVOICE_DATE) AS month,
    SUM(t.QUANTITY * t.transactions_PRICE) AS total_sales
FROM 
    Transactions t
JOIN 
    Invoice i ON t.INVOICE_ID = i.INVOICE_ID
GROUP BY 
    year, month
ORDER BY 
    year DESC, month DESC;

2. Top 5 Best-Selling Products by Quantity
This query identifies the top 5 products by the total quantity sold, using the Transactions and Product tables.

SELECT 
    p.PRODUCT_NAME,
    SUM(t.QUANTITY) AS total_quantity_sold
FROM 
    Transactions t
JOIN 
    Product p ON t.PRODUCT_SKU = p.PRODUCT_SKU
GROUP BY 
    p.PRODUCT_NAME
ORDER BY 
    total_quantity_sold DESC
LIMIT 5;

3. Sales by Customer Price Tier
This query shows total sales based on different customer price tiers, using the Transactions, Invoice, and Customer tables.

SELECT 
    c.PRICE_TIER,
    SUM(t.QUANTITY * t.transactions_PRICE) AS total_sales
FROM 
    Transactions t
JOIN 
    Invoice i ON t.INVOICE_ID = i.INVOICE_ID
JOIN 
    Customer c ON i.CUSTOMER_ID = c.CUSTOMER_ID
GROUP BY 
    c.PRICE_TIER
ORDER BY 
    total_sales DESC;

4. Total Sales by Product Category
This query shows the total revenue generated by each product category, using the Transactions, Product, and Category tables.

SELECT 
    cat.CATEGORY_NAME,
    SUM(t.QUANTITY * t.transactions_PRICE) AS total_sales
FROM 
    Transactions t
JOIN 
    Product p ON t.PRODUCT_SKU = p.PRODUCT_SKU
JOIN 
    Category cat ON p.CATEGORY_ID = cat.CATEGORY_ID
GROUP BY 
    cat.CATEGORY_NAME
ORDER BY 
    total_sales DESC;

5. Inventory Level by Location
This query gives an overview of the inventory levels per product at each store location, using the Inventory and Location tables.

SELECT 
    loc.LOCATION_NAME,
    p.PRODUCT_NAME,
    i.INVENTORY_QUANTITY
FROM 
    Inventory i
JOIN 
    Product p ON i.PRODUCT_SKU = p.PRODUCT_SKU
JOIN 
    Location loc ON i.LOCATION_ID = loc.LOCATION_ID
ORDER BY 
    loc.LOCATION_NAME, p.PRODUCT_NAME;

6. Average Sale Amount per Invoice
This query calculates the average sale amount per invoice using the Invoice and Transactions tables.

SELECT 
    AVG(i.INVOICE_TOTAL) AS average_invoice_amount
FROM 
    Invoice i
JOIN 
    Transactions t ON i.INVOICE_ID = t.INVOICE_ID;
